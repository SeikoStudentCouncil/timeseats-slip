\begin{luacode*}
  function generateCheckDigit(number)
    local str = string.format("%05d", number)
    local digits = {}

    for i = 1, #str do
      digits[i] = tonumber(str:sub(i,i))
    end

    local rev = {}
    for i = #digits, 1, -1 do
      rev[#digits - i + 1] = digits[i]
    end
    
    local sum = 0
    for i = 1, #rev do
      local n = rev[i]

      if ((i-1) % 2) == 0 then
        n = n * 2
        if n > 9 then n = n - 9 end
      end
      sum = sum + n
    end
    
    local checkDigit = (10 - (sum % 10)) % 10
    return checkDigit
  end
\end{luacode*}
